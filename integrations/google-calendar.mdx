<Steps>

  </Step>

  <Step>
    <h3>Presiona en Conectar</h3>

  <Step>
    <h3>Conecta la app de Doko con tu cuenta de Google</h3>

  </Step>

  <Step>
    <h3>Da los permisos necesarios para que Doko pueda acceder a tu cuenta de Google</h3>

    <Frame>

    </Frame>
  </Step>

  <Step>
    <h3>Listo! Tu chatbot ya puede agendar citas en tu cuenta de Google Calendar </h3>

  </Step>
